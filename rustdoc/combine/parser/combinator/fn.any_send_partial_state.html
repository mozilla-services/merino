pub fn any_send_partial_state<Input, P>(p: P) -> AnySendPartialStateParser<P> where
    Input: Stream,
    P: Parser<Input>,
    P::PartialState: Send + 'static, 
Expand description

Returns a parser where P::PartialState is boxed. Useful as a way to avoid writing the type since it can get very large after combining a few parsers.



parser! {
    type PartialState = AnySendPartialState;
    fn example[Input]()(Input) -> (char, char)
    where [ Input: Stream<Token = char> ]
    {
        any_send_partial_state((letter(), letter()))
    }
}

assert_eq!(
    example().easy_parse("ab"),
    Ok((('a', 'b'), ""))
);
